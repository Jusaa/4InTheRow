62      public void voitonTarkistaja(int nro) {
63          String voitto = "";
64  
65          //Pystyrivien tarkistus
66          pystyTarkistus(nro, voitto);
67  
68          //Vaakarivien tarkistus
69          vaakaTarkistus(nro, voitto);
70  
71          //Vinorivien tarkistus OIKEALLE YLÖS
72          vinoTarkistusOY(nro, voitto, 4, 2);
73          vinoTarkistusOY(nro, voitto, 5, 1);
74          vinoTarkistusOY(nro, voitto, 6, 0);
75          vinoTarkistusOY(nro, voitto, 6, -1);
76          vinoTarkistusOY(nro, voitto, 5, -2);
77          vinoTarkistusOY(nro, voitto, 4, -3);
78  
79          //Vinorivien tarkistus OIKEALLE ALAS       
80          vinoTarkistusOA(nro, voitto, 4, 3, 0);
81          vinoTarkistusOA(nro, voitto, 5, 4, 0);

83          vinoTarkistusOA(nro, voitto, 6, 5, 1);
84          vinoTarkistusOA(nro, voitto, 5, 5, 2);

89      public void dingDingDing(int nro, String voitto) {
90          if (voitto.equals("1111")) {
91              System.out.println("DINGDINGDING\nPELAAAJA " + nro + " VOITTI!");
92              tarkistusnro = nro;
93          }
94      }

96      public void vinoTarkistusOY(int nro, String voitto, int maara, int getOffset) {
97          voitto = "";
98          for (int i = 0; i < maara; i++) {
99              if (getOffset < 0) {
100                 if (piirtaja.getLista().get(i - getOffset).get(i) == nro) {
101                     voitto = voitto + "1";
102                 } else {
103                     voitto = "";
104                 }
105             } else {
106                 if (piirtaja.getLista().get(i).get(i + getOffset) == nro) {
107                     voitto = voitto + "1";
108                 } else {
109                     voitto = "";
110                 }
111             }
112 
113             dingDingDing(nro, voitto);
114         }
115     }

117     public void vinoTarkistusOA(int nro, String voitto, int maara, int laskeva, int nouseva) {
118         voitto = "";
119         int ii = laskeva;
120         for (int i = nouseva; i < maara; i++) {
121             if (piirtaja.getLista().get(i).get(ii) == nro) {
122                 voitto = voitto + "1";
123             } else {
124                 voitto = "";
125             }
126             ii--;
127             dingDingDing(nro, voitto);
128         }
129     }

131     public void vaakaTarkistus(int nro, String voitto) {
132         for (int i = 0; i < 6; i++) {
133             voitto = "";
134             for (int ii = 0; ii < 7; ii++) {
135                 if (piirtaja.getLista().get(ii).get(i) == nro) {
136                     voitto = voitto + "1";
137                 } else {
138                     voitto = "";
139                 }
140                 dingDingDing(nro, voitto);
141             }
142         }
143     }

145     public void pystyTarkistus(int nro, String voitto) {
146         voitto = "";
147         for (int i = 6; i >= 0; i--) {
148             voitto = "";
149             for (int ii = 0; ii < 6; ii++) {
150                 if (piirtaja.getLista().get(i).get(ii) == nro) {
151                     voitto = voitto + "1";
152                 } else {
153                     voitto = "";
154                 }
155                 dingDingDing(nro, voitto);
156             }
157         }
158     }
