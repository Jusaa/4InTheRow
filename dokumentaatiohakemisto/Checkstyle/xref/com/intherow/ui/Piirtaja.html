53      /**
54       * Metodi alustaa uuden pelin
55       *
56       * @throws IOException if MouseListener fails
57       */
58      public void uusiPeli() throws IOException {
59  
60          database.uusiLista();
61          alusta.getContentPane().removeAll();
62  
63          for (int i = 1; i < 8; i++) {
64              alusta.getContentPane().add(new JButton("" + i));
65              alusta.getContentPane().getComponent(i - 1).addMouseListener(new HiirenKuuntelija(i, this));
66          }
67          for (int i = 0; i < 42; i++) {
68              alusta.getContentPane().add(new JLabel(tyhjaruutu));
69          }
70          alusta.pack();
71          alusta.setVisible(true);
72      }

74      /**
75       * Metodi lisää pelikentälle uuden merkin
76       *
77       * @param pylvas mihin pylvääseen merkki laitetaan
78       * @param merkki pelaajan oma merkki
79       * @param merkki2 vastustajan merkki
80       * @param id pelaajan oma id arvo
81       * @return lisättiinkö uusi merkki vai ei
82       */
83      public boolean vuoronPiirto(int pylvas, String merkki, String merkki2, int id) {
84  
85          if (database.lisaaListaan(pylvas, merkki, merkki2, id) == false) {
86              return false;
87          } else {
88              int ruutuId = 0;
89              int nro = 6;
90              for (int i = nro; i > 0; i--) {
91                  for (int ii = 0; ii < 7; ii++) {
92                      if (database.getLista().get(ii).get(nro - 1) == 1 && id == 1) {
93                          ruudunMuokkaus(ruutuId + 7, id);
94                      } else if (database.getLista().get(ii).get(nro - 1) == 2 && id == 2) {
95                          ruudunMuokkaus(ruutuId + 7, id);
96                      }
97  
98                      ruutuId++;
99                  }
100                 nro--;
101 
102             }
103             alusta.validate();
104             return true;
105         }
106     }

108     /**
109      * Metodi piirtää menun jonka avulla pelaaja päättää miten jatketaan
110      *
111      * @param lukija scanner joka lukee pelaajan inputin
112      * @return valinta miten jatketaan peliä
113      */
114     public int menuPiirto(Scanner lukija) {
115         while (true) {
116             System.out.println("\n[1] Uusi yksinpeli\n[2] Uusi kaksinpeli\n[3] Uusi peli samoilla tiedoilla\n[4] Lopeta");
117             String komento = lukija.nextLine();
118             if (komento.equals("1")) {
119                 return 1;
120             } else if (komento.equals("2")) {
121                 return 2;
122             } else if (komento.equals("3")) {
123                 return 3;
124             } else if (komento.equals("4")) {
125                 return 4;
126             }
127         }
128     }

136     public void ruudunMuokkaus(int ruutuId, int id) {
137         alusta.getContentPane().remove(ruutuId);
138         if (id == 1) {
139             alusta.getContentPane().add(new JLabel(pelaajan1ruutu), ruutuId);
140         } else if (id == 2) {
141             alusta.getContentPane().add(new JLabel(pelaajan2ruutu), ruutuId);
142         }
143 
144     }
