22      public int tarkasta(int nro, Piirtaja piirtaja) {
23          String voitto = "";
24  
25          //Pystyrivien tarkistus
26          pystyTarkistus(nro, voitto, piirtaja);
27  
28          //Vaakarivien tarkistus
29          vaakaTarkistus(nro, voitto, piirtaja);
30  
31          //Vinorivien tarkistus OIKEALLE YLÃ–S
32          vinoTarkistusOY(nro, voitto, 4, 2, piirtaja);
33          vinoTarkistusOY(nro, voitto, 5, 1, piirtaja);
34          vinoTarkistusOY(nro, voitto, 6, 0, piirtaja);
35          vinoTarkistusOY(nro, voitto, 6, -1, piirtaja);
36          vinoTarkistusOY(nro, voitto, 5, -2, piirtaja);
37          vinoTarkistusOY(nro, voitto, 4, -3, piirtaja);
38  
39          //Vinorivien tarkistus OIKEALLE ALAS       
40          vinoTarkistusOA(nro, voitto, 4, 3, 0, piirtaja);
41          vinoTarkistusOA(nro, voitto, 5, 4, 0, piirtaja);
42          vinoTarkistusOA(nro, voitto, 6, 5, 0, piirtaja);
43          vinoTarkistusOA(nro, voitto, 6, 5, 1, piirtaja);
44          vinoTarkistusOA(nro, voitto, 5, 5, 2, piirtaja);
45          vinoTarkistusOA(nro, voitto, 4, 5, 3, piirtaja);
46  
47          if (tarkistusnro == 0) {
48              int apunumero = 0;
49              for (int i = 0; i < 7; i++) {
50                  if (!piirtaja.getDatabase().getLista().get(i).contains(0)) {
51                      apunumero++;
52                  }
53                  if (apunumero == 7) {
54                      System.out.println("TASAPELI!");
55                      tarkistusnro = 3;
56                  }
57              }
58          }
59  
60          return tarkistusnro;
61      }

63      public void dingDingDing(int nro, String voitto) {
64          if (voitto.equals("1111")) {
65              System.out.println("DINGDINGDING\nPELAAAJA " + nro + " VOITTI!");
66              tarkistusnro = nro;
67          }
68      }

70      public void vinoTarkistusOY(int nro, String voitto, int maara, int getOffset, Piirtaja piirtaja) {
71          voitto = "";
72          for (int i = 0; i < maara; i++) {
73              if (getOffset < 0) {
74                  if (piirtaja.getDatabase().getLista().get(i - getOffset).get(i) == nro) {
75                      voitto = voitto + "1";
76                  } else {
77                      voitto = "";
78                  }
79              } else {
80                  if (piirtaja.getDatabase().getLista().get(i).get(i + getOffset) == nro) {
81                      voitto = voitto + "1";
82                  } else {
83                      voitto = "";
84                  }
85              }
86  
87              dingDingDing(nro, voitto);
88          }
89      }
90  
91      public void vinoTarkistusOA(int nro, String voitto, int maara, int laskeva, int nouseva, Piirtaja piirtaja) {
92          voitto = "";
93          int ii = laskeva;
94          for (int i = nouseva; i < maara + nouseva; i++) {
95              if (piirtaja.getDatabase().getLista().get(i).get(ii) == nro) {
96                  voitto = voitto + "1";
97              } else {
98                  voitto = "";
99              }
100             ii--;
101             dingDingDing(nro, voitto);
102         }
103     }

105     public void vaakaTarkistus(int nro, String voitto, Piirtaja piirtaja) {
106         for (int i = 0; i < 6; i++) {
107             voitto = "";
108             for (int ii = 0; ii < 7; ii++) {
109                 if (piirtaja.getDatabase().getLista().get(ii).get(i) == nro) {
110                     voitto = voitto + "1";
111                 } else {
112                     voitto = "";
113                 }
114                 dingDingDing(nro, voitto);
115             }
116         }
117     }
118 
119     public void pystyTarkistus(int nro, String voitto, Piirtaja piirtaja) {
120         voitto = "";
121         for (int i = 6; i >= 0; i--) {
122             voitto = "";
123             for (int ii = 0; ii < 6; ii++) {
124                 if (piirtaja.getDatabase().getLista().get(i).get(ii) == nro) {
125                     voitto = voitto + "1";
126                 } else {
127                     voitto = "";
128                 }
129                 dingDingDing(nro, voitto);
130             }
131         }
132     }
