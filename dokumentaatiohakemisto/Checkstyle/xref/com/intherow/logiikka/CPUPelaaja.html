1   package com.intherow.logiikka;
2   
3   import com.intherow.ui.Piirtaja;
4   import java.util.Random;
5   import java.util.Scanner;
6   
7   /**
8    *
9    * @author Jusaa
10   */
11  public class CPUPelaaja implements Pelaaja {
12  
13      Random random;
14      private String nimi;
15      private int id;
16      private String pelaajanMerkki;
17      private String vastustajanMerkki;
18      private Piirtaja piirtaja;
19      private int voitot;
20  
21      public CPUPelaaja(String nimi, String pelaajanMerkki, Piirtaja piirtaja, int id) {
22          this.nimi = nimi;
23          this.piirtaja = piirtaja;
24          this.pelaajanMerkki = pelaajanMerkki;
25          this.id = id;
26          voitot = 0;
27          random = new Random();
28          //this.vastustajanMerkki = null;
29      }
30  
31      /**
32       * Metodi laittaa merkin cpupelaajan tahdon mukaisesti
33       *
34       * @param piirtaja hakee alustan mihin laittaa merkin
35       */
36      public void laitaMerkki(Piirtaja piirtaja) {
37          //Graafiseen liittymään siirtymisen jälkeen tämän parantaminen
38          while (true) {
39              int pylvas = random.nextInt(6) + 1;
40              if (piirtaja.vuoronPiirto(pylvas, pelaajanMerkki, vastustajanMerkki, id) == true) {
41                  break;
42              }
43          }
44      }
45  
46      /**
47       * Metodi asettaa vastustajan merkin
48       *
49       * @param merkki asettaa vastustajan merkin jotta piirtäminen onnistuu
50       */
51      public void asetaVastustajanMerkki(String merkki) {
52          vastustajanMerkki = merkki;
53      }
54  
55      /**
56       * lisää voittojen määrää
57       */
58      public void voitti() {
59          voitot++;
60      }
61  
62      public String getNimi() {
63          return nimi;
64      }
65  
66      public String getOmaMerkki() {
67          return pelaajanMerkki;
68      }
69  
70      public String getVastustajanMerkki() {
71          return vastustajanMerkki;
72      }
73  
74      public int getVoitot() {
75          return voitot;
76      }
77  
78      @Override
79      public Piirtaja getPiirtaja() {
80          return piirtaja;
81      }
82  
83      @Override
84      public String toString() {
85          if (vastustajanMerkki != null) {
86              return nimi + ", oma merkki " + pelaajanMerkki + ", vastustajan merkki " + vastustajanMerkki;
87          }
88          return nimi + ", oma merkki " + pelaajanMerkki;
89      }
90  
91  }
